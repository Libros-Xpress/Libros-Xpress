 16def crear_portada_placeholder(titulo, ancho=160, alto=210):
 17    pix = QPixmap(ancho, alto)
 18    pix.fill(QColor("#F5E1C0"))
 19    painter = QPainter(pix)
 20    painter.setRenderHint(QPainter.Antialiasing)
 21    pen = QPen(QColor("#D4A574"), 2)
 22    painter.setPen(pen)
 23    painter.drawRoundedRect(2, 2, ancho-4, alto-4, 10, 10)
 24    font_icono = QFont("Segoe UI Emoji", 30)
 25    painter.setFont(font_icono)
 26    painter.setPen(QColor("#8B5E3C"))
 27    painter.drawText(pix.rect(), Qt.AlignHCenter | Qt.AlignTop, "\n\n📖")
 28    font_titulo = QFont("Segoe UI", 11, QFont.Bold)
 29    painter.setFont(font_titulo)
 30    painter.setPen(QColor("#5D4037"))
 31    painter.drawText(pix.rect().adjusted(8, 70, -8, -8),
 32                     Qt.AlignHCenter | Qt.TextWordWrap, titulo)
 33    painter.end()
 34    return pix

173            color_stock = "#2E7D32" if stock > 0 else "#C0392B"

175            lbl_stock = QLabel(texto_stock)

177            lbl_stock.setStyleSheet(f"font-size: 11px; color: {color_stock};")
178            ly.addWidget(lbl_stock)
179
180            # Botón Agregar
181            btn = QPushButton("Agregar al carrito")
182            if stock <= 0:
183                btn.setEnabled(False)
184                btn.setStyleSheet("""
185                    QPushButton { background-color: #E0E0E0; color: #9E9E9E;
186                                  font-weight: bold; border: none; padding: 6px;
187                                  border-radius: 6px; font-size: 12px; }

194                    QPushButton:hover { background-color: #81C784; }
195                """)
196                if on_agregar:
197                    btn.clicked.connect(lambda _, t=prod.titulo, p=prod.precio: on_agregar(t, p))
198            ly.addWidget(btn)

200            self.scroll_layout.addWidget(tarjeta, fila, col)
